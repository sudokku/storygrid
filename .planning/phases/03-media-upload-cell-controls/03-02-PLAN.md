const undo = useGridStore(s => s.undo);

useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    // Only fire if not typing in an input/textarea
    const target = e.target as HTMLElement;
    if (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA' || target.isContentEditable) return;

    if (e.ctrlKey && e.key === 'z' && !e.shiftKey) {
      e.preventDefault();
      undo();
    } else if (e.ctrlKey && e.shiftKey && e.key === 'z') {
      e.preventDefault();
      redo();
    }
  };
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [undo, redo]);

**Helper: compute cell pixel dimensions**

```typescript
/**
 * Computes the pixel dimensions of a leaf node at 1080×1920 resolution.
 * Walks from root to target, multiplying fractional sizes at each container level.
 */
function computeCellDimensions(root: GridNode, nodeId: string): { w: number; h: number } | null {
  let wFraction = 1;
  let hFraction = 1;

  function walk(node: GridNode, targetId: string): boolean {
    if (node.id === targetId) return true;
    if (node.type !== 'container') return false;
    const totalWeight = node.sizes.reduce((sum, s) => sum + s, 0);
    for (let i = 0; i < node.children.length; i++) {
      const childFraction = node.sizes[i] / totalWeight;
      // Temporarily apply fraction
      const prevW = wFraction;
      const prevH = hFraction;
      if (node.direction === 'horizontal') {
        hFraction *= childFraction;
      } else {
        wFraction *= childFraction;
      }
      if (walk(node.children[i], targetId)) return true;
      // Restore if not on this path
      wFraction = prevW;
      hFraction = prevH;
    }
    return false;
  }

  if (!walk(root, nodeId)) return null;
  return {
    w: Math.round(1080 * wFraction),
    h: Math.round(1920 * hFraction),
  };
}
```

Wait — the recursive restore approach above has a bug (restoring after recursive call returns false). Use a cleaner path-based approach instead:

```typescript
function computeCellDimensions(root: GridNode, nodeId: string): { w: number; h: number } | null {
  // Find path from root to target as array of [node, childIndex] pairs
  function findPath(node: GridNode, targetId: string, path: Array<{ node: ContainerNode; idx: number }>): boolean {
    if (node.id === targetId) return true;
    if (node.type !== 'container') return false;
    for (let i = 0; i < node.children.length; i++) {
      path.push({ node: node as ContainerNode, idx: i });
      if (findPath(node.children[i], targetId, path)) return true;
      path.pop();
    }
    return false;
  }

  const path: Array<{ node: ContainerNode; idx: i }> = [];
  // Fix: idx type is number
  const typedPath: Array<{ node: ContainerNode; idx: number }> = [];
  if (!findPath(root, nodeId, typedPath)) return null;

  let wFraction = 1;
  let hFraction = 1;
  for (const { node, idx } of typedPath) {
    const totalWeight = node.sizes.reduce((sum, s) => sum + s, 0);
    const fraction = node.sizes[idx] / totalWeight;
    if (node.direction === 'vertical') {
      // vertical container stacks children top-to-bottom → splits height
      hFraction *= fraction;
    } else {
      // horizontal container places children side-by-side → splits width
      wFraction *= fraction;
    }
  }
  return { w: Math.round(1080 * wFraction), h: Math.round(1920 * hFraction) };
}
```

IMPORTANT: In this project, `direction: 'vertical'` means children are arranged top-to-bottom (vertical stack → splits height). `direction: 'horizontal'` means children are side-by-side (horizontal row → splits width). Verify this matches how `ContainerNode.tsx` renders (it uses `flexDirection: direction === 'vertical' ? 'column' : 'row'`).

describe('D-04A: AudioContext pre-creation before first await', () => {
  let audioContextConstructorCallOrder: number;
  let getCodecCallOrder: number;
  let callCounter: number;

    callCounter = 0;
    audioContextConstructorCallOrder = -1;
    getCodecCallOrder = -1;

    vi.restoreAllMocks();

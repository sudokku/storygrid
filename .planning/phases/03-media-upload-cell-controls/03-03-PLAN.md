```typescript
// Mock FileReader to return a controlled data URI
function mockFileReader(result: string | null, error?: DOMException) {
  const originalFileReader = global.FileReader;
  const mockFR = {
    readAsDataURL: vi.fn(function(this: any) {
      setTimeout(() => {
        if (error) { this.onerror?.(new ProgressEvent('error')); }
        else { this.result = result; this.onload?.(new ProgressEvent('load')); }
      }, 0);
    }),
    result: null as string | null,
    onload: null as ((e: ProgressEvent) => void) | null,
    onerror: null as ((e: ProgressEvent) => void) | null,
  };
  vi.spyOn(global, 'FileReader' as any).mockImplementation(() => mockFR);
  return { mockFR, restore: () => { global.FileReader = originalFileReader; } };
}
```

**2. Add pinch-to-zoom touch handlers to `src/Grid/LeafNode.tsx` (per D-11):**

Add two refs at the top of the component:
```typescript
const pinchStartDistRef = useRef<number>(0);
const pinchStartScaleRef = useRef<number>(1);
```

Add a `useEffect` that registers native touch event listeners on the leaf div (NOT the canvas) when `isPanMode` is true. This is needed because `touchmove` must be registered with `{ passive: false }` to call `preventDefault()`, which React's synthetic event system does not support:

```typescript
useEffect(() => {
  if (!isPanMode) return;
  const el = divRef.current; // the leaf wrapper div
  if (!el) return;

  const handleTouchStart = (e: TouchEvent) => {
    if (e.touches.length === 2) {
      pinchStartDistRef.current = Math.hypot(
        e.touches[1].clientX - e.touches[0].clientX,
        e.touches[1].clientY - e.touches[0].clientY
      );
      pinchStartScaleRef.current = node?.panScale ?? 1;
    }
  };

  const handleTouchMove = (e: TouchEvent) => {
    if (e.touches.length !== 2) return;
    e.preventDefault(); // prevent page scroll during pinch
    const dist = Math.hypot(
      e.touches[1].clientX - e.touches[0].clientX,
      e.touches[1].clientY - e.touches[0].clientY
    );
    const ratio = dist / pinchStartDistRef.current;
    const newScale = Math.max(1, Math.min(3, pinchStartScaleRef.current * ratio));
    updateCell(id, { panScale: newScale });
  };

  const handleTouchEnd = () => {
    pinchStartDistRef.current = 0;
    pinchStartScaleRef.current = 1;
  };

  el.addEventListener('touchstart', handleTouchStart, { passive: true });
  el.addEventListener('touchmove', handleTouchMove, { passive: false });
  el.addEventListener('touchend', handleTouchEnd);

  return () => {
    el.removeEventListener('touchstart', handleTouchStart);
    el.removeEventListener('touchmove', handleTouchMove);
    el.removeEventListener('touchend', handleTouchEnd);
  };
}, [isPanMode, id, node?.panScale, updateCell]);
```

Note: `divRef` is the ref on the outermost wrapper div of the leaf. Check if it already exists — the component likely has a ref for the canvas element. If `divRef` does not exist, add a new ref for the leaf wrapper div. The touch events must be on the wrapper div (not the canvas child) so they capture two-finger gestures anywhere in the cell.
